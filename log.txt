//例题：求解最长公共子序列（lcs）
#include<iostream>
using namespace std;
 
const int MAXN = 101;
 
int lcs(char* s1, char* s2, int n1, int n2)
{
    // 构建dp表，长度为n1 + 1，宽度为n2 + 1
    int dp[MAXN][MAXN];
 
    // lcs(i, j)表示s1[1..i]与s2[1..j]的最长公共子序列
    for(int i = 0; i <= n1; i++)
    {
        for(int j = 0; j <= n2; j++)
        {
            if(i == 0 || j == 0)
                dp[i][j] = 0;
            else if(s1[i-1] == s2[j-1])
                dp[i][j] = dp[i-1][j-1] + 1;
            else
                dp[i][j] = max(dp[i-1][j], dp[i][j-1]);
        }
    }
    return dp[n1][n2];
}
